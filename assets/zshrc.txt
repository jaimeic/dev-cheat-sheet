## Usage: gbdel branch_name

    # Branch name present?
    if [[ -z "$1" ]]; then
        echo "\nðŸ¤” Oopsâ€¦ you forgot to provide the branch name"

    else

        echo "\nâ³ Deletingâ€¦\n"
        git branch -D "$1" # Local delete.
        git push origin --delete "$1" # Remote delete.
        echo "\nâœ…âœ…Git branch $1 was deleted from local and remote.\n"
    fi
}

## Create branch locally and remotely
## Usage: gbcreate branch_name
function gbcreate {
    # Branch name present?
    if [[ -z "$1" ]]; then
        echo "\nðŸ¤” Oopsâ€¦ you forgot to provide the branch name"
        echo "ðŸ‘‰ e.g. gbcreate branch_name\n"
    else
        # Start creating branch.
        echo "\nâ³ Creatingâ€¦\n"
        git checkout -b "$1" # Local create.
        git push -u origin "$1" # Remote create.
        echo "\nâœ… Git branch $1 was created in local and remote.\n"
    fi
}
